import cv2
import matplotlib.pyplot as plt

def so_khop_dac_trung(img1, img2, k=2, ratio_thresh=0.75):
    """
    Hm so khp c trng gia hai nh bng SIFT, Brute-Force Matcher, v ratio test.

    Tham s:
    img1: nh mu (template image)
    img2: nh gc (query image)
    k: s lng im nh hng xm gn nht  kim tra
    ratio_thresh: ngng  lc cc im khp tt (thng ly gi tr t 0.7 n 0.8)

    Tr v:
    matches: danh sch cc im khp tt
    """
    # Chuyn nh sang thang mu xm
    gray1 = cv2.cvtColor(img1, cv2.COLOR_BGR2GRAY)
    gray2 = cv2.cvtColor(img2, cv2.COLOR_BGR2GRAY)
    
    # Khi to SIFT detector
    sift = cv2.SIFT_create()
    
    # Tm cc im c trng v m t c trng
    keypoints1, descriptors1 = sift.detectAndCompute(gray1, None)
    keypoints2, descriptors2 = sift.detectAndCompute(gray2, None)
    
    # To BF matcher vi L2 norm, chn false  khng sp xp cc im khp
    bf = cv2.BFMatcher(cv2.NORM_L2, crossCheck=False)
    
    # Tm k im khp tt nht
    matches = bf.knnMatch(descriptors1, descriptors2, k=k)
    
    # Lc cc im khp bng ratio test
    good_matches = []
    for m, n in matches:
        if m.distance < ratio_thresh * n.distance:
            good_matches.append(m)

    return keypoints1, keypoints2, good_matches

# c nh mu v nh gc
img1 = cv2.imread('image/church1.png')  # ng dn ti nh mu
img2 = cv2.imread('image/church2.png')      # ng dn ti nh gc

# Thc hin so khp
keypoints1, keypoints2, good_matches = so_khop_dac_trung(img1, img2)

# V cc im khp tt ln nh  trc quan
img_matches = cv2.drawMatches(img1, keypoints1, img2, keypoints2, good_matches, 
                              None, flags=cv2.DrawMatchesFlags_NOT_DRAW_SINGLE_POINTS)

# Hin th nh kt qu
plt.imshow(cv2.cvtColor(img_matches, cv2.COLOR_BGR2RGB))
plt.show()
